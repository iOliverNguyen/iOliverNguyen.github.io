We can use this pattern to output log with color in browser by adding CSS:

const logger = (name) => ({
  info: console.info.bind(console, `%c${name}: %s`, 'color: #56d74b'),
  verbose: console.debug.bind(console, `%c${name}: %s`, 'color: #70818e'),
})
